MethodBeforeAdvice
------------------
class LoanManager {
	public boolean approveLoan(long loanNo) {
		sop("in approveLoan()");
		return true;
	}
}

class AuditAspect {
	public void doAudit(Joinpoint jp) {
		String methodName = null;
		Object[] args = null;
		
		methodName = jp.getSignature().getName();
		args = jp.getArgs();
		sop(methodName+"("+args[0]+") has been called by Robert at : " + LocalDateTime.now());		
	}
}

application-context.xml
-----------------------
<bean id="loanManager" class="LoanManager"/>
<bean id="auditAspect" class="AuditAspect"/>

<aop:config>
	<aop:aspect ref="auditAspect">
		<aop:before method="doAudit" pointcut="execution(returnType pkg.className.methodName(args))"/>
	</aop:aspect>
</aop:config>
